\makeproblem{Convolution theorem}{pr:convolution:1}{

Find the form of the Fourier transform

\begin{dmath}\label{eqn:convolution:10}
\tilde{f}(\omega) = \int f(y) e^{-i \omega y} dy
\end{dmath}

of a convolution integral

\begin{dmath}\label{eqn:convolution:30}
f(y) = \int dy' g(y') h(y - y') = g(y) \conj h(y).
\end{dmath}

} % makeproblem

\makeanswer{pr:convolution:1}{

This is basically just an application of change of variables

\begin{dmath}\label{eqn:convolution:50}
\calF f(y)
=
\int dy e^{-i \omega y}
dy' g(y') h(y - y')
=
\int dy' g(y')
\int dy e^{-i \omega y} \mathLabelBox{h(y - y')}{
\(y - y' = u\)
}
=
\int dy' g(y')
\int du e^{-i \omega (u + y')} h(u)
=
\tilde{h}(\omega)
\int dy' g(y') e^{-i \omega y'}
=
\tilde{h}(\omega)
\tilde{g}(\omega)
\end{dmath}

} % makeanswer
